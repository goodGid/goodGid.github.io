## [1] Answer Code (18. 02. 05)
``` cpp
#include<iostream>
using namespace std;

int main(){
    int n,k;
    cin >> n >> k;
    
    if(k==0)
        cout << n << endl;
    else{
        int tmp = 1;
        for(int i=0; i<k; i++)
            tmp *= 10;
        k = tmp;
        int a = n % k ;
        if( a >= k/2)
            cout << ( n/k + 1) * k << endl;
        else
            cout << ( n/k ) * k << endl;
    }
    
    return 0;
}
```

## [2] Answer Code (18. 02. 05)
``` cpp

#include<iostream>
using namespace std;
    
int C, K, t = 1;
int main() {
	scanf("%d %d", &C, &K);
	for (int i = 0; i < K; i++) t *= 10;
	K = t / 10;
	printf("%d\n", (C + (K * 5)) / (K * 10) * (K * 10));

	return 0;
}
```
