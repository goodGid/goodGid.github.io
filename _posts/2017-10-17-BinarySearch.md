## Code
``` cpp

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;
vector<int> v;

int main(){
    int n, tc, value;
    cin >> n;
    for(int i=0; i<n; i++){
        scanf("%d",&value);
        v.push_back(value);
    }
    sort(v.begin(),v.end());
    
    cin >> tc;
    for(int i=0; i<n; i++){
        scanf("%d",&value);
        
        if( binary_search(v.begin(), v.end(), value) )
            printf("Exist\n");
        else
            printf("Not Exist\n");
    }
    return 0;
}


```
