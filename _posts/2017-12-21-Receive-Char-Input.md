## Code

``` cpp

#include<iostream>
using namespace std;

int main(){
    int n;
    cin >> n;
    
    for(int i=0; i<n; i++){
        char m,l,r;

        // To Do
        // A B C를 입력하고 싶다.


        /*
        Case 1 
        3
        A B C

        --> 1st '%c'를 1칸 Space해줘야 한다.
        */
        scanf(" %c %c %c",&m,&l,&r);

        /*
        Case 2
        3ABC
        */
        scanf("%c%c%c",&m,&l,&r);


        /*
        Case 3
        3A
        B C
        */
        scanf("%c %c %c",&m,&l,&r);

        /*
        Case 4
        3
        ABC
        m = '\n'
        l = 'A'
        r = 'B' 
        이렇게 들어간다.
        */
        scanf("%c %c %c",&m,&l,&r);

    }
    return 0;
}

```
