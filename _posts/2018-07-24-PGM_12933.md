## [1] Answer Code (18. 07. 24)

``` cpp

// [1]
long long solution(long long n) {
    long long answer = 0;
    string s = "";
    vector<long long> v;
    
    while (n != 0) {
        v.push_back(n % 10);
        n /= 10;
    }
    sort(v.rbegin(), v.rend());
    for(int i=0; i<v.size(); i++){
        s += to_string(v[i]);
    }
    answer = atoll(s.c_str());
    
    
    return answer;
}


// [2]
```
