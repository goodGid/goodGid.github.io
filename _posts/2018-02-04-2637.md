## Answer Code
``` cpp
#include<iostream>
#include<vector>
#include<queue>
using namespace std;

vector<vector<int>> v;
queue<int> q;
int d[101][101]; // d is DP
int visit[101]; // v is Visit
int chk[101];

int main(){
    int n,m;
    cin >> n >> m;

    v.resize(n+1);
    while (m--) {
        int a,b,c;
        scanf("%d%d%d",&a,&b,&c);
        chk[a] ++ ;
        v[b].push_back(a);
        d[a][b] = c;        // a를 만들기 위해 b부품 c개 필요
    }
    for(int i=1; i<=n; i++){
        if(chk[i] == 0){
            visit[i] = 1;
            q.push(i);
        }
    }
    
    while (! q.empty()) {
        int top = q.front();
        q.pop();
        
        int size = (int) v[top].size();
        for(int i=0; i<size; i++){
            int nx = v[top][i];
            chk[nx] --;
            
            if(chk[nx] == 0){
                q.push(nx);
            }
            for(int j=1; j<=n && visit[top] != 1; j++){
                    d[nx][j] += ( d[top][j] * d[nx][top]);
            }
        }
    }
    
    for (int i=1; i<=n; i++) {
        if(d[n][i] != 0 && visit[i] == 1)
            printf("%d %d\n",i,d[n][i]);
    }
    
    return 0;
}
```
