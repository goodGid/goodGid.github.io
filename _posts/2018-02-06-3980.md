## Answer Code 
``` cpp

#include<iostream>
#include<algorithm>
using namespace std;

int map[12][12];
int pos_y[12];

int dfs(int x,int value){
    if( x > 11 ) return value;
    int ans = 0;
    for(int i=1; i<=11; i++){
        if(map[x][i] != 0 && !pos_y[i]){
            pos_y[i] = 1;
            ans = max( ans, dfs(x+1, value + map[x][i]) ) ;
            pos_y[i] = 0;
        }
    }
    return ans;
}

int main(){
    int tc;
    cin >> tc;
    while (tc--) {
        for(int i=1; i<=11; i++){
            for(int j=1; j<=11; j++){
                scanf("%d",&map[i][j]);
            }
        }
        int ans = 0;
        for(int i=1; i<=11; i++){
            if( map[1][i] ){
                pos_y[i] = 1;
                ans = max ( ans, dfs(2 ,map[1][i]));
                pos_y[i] = 0;
            }
        }
        printf("%d\n",ans);
    }
    
    return 0;
}
```
