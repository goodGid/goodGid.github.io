## [1] Answer Code (18. 03. 21)

``` cpp

#include<iostream>
using namespace std;

int c[101]; // c is coin
int dp[10001];

int main(){
    int n,k;
    cin >> n >> k;
    
    for(int i=1; i<=n; i++)
        scanf("%d", c+i);
    
    dp[0] = 1;
    
    for(int i=1; i<=n; i++){
        for(int j=c[i]; j<=k; j++){
                dp[j] += dp[ j - c[i]];
        }
    }
    cout << dp[k] << endl;
    
    return 0;
}
```
