## [1] Answer Code (18. 10. 19)

``` cpp
#include<iostream>
#include<vector>
#define p pair<int,int>
using namespace std;

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    
    int n;
    cin >> n;
    
    vector<int> v;
    for(int i=0; i<n; i++){
        int tmp;
        cin >> tmp;
        v.push_back(tmp);
    }
    
    int a,b;
    cin >> a >> b;
    long long cnt = 0;
    for(int i=0; i<n; i++){
        int student = v[i];
        student -= a;
        if( student < 0 )
            continue;
        if( student % b == 0)
            cnt += student / b;
        else
            cnt += student / b + 1;
    }
    
    cnt += n;
    cout << cnt << endl;
    
    return 0;
}
```
