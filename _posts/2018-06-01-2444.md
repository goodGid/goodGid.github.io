## [1] Answer Code (18. 06. 01)


``` cpp

#include<iostream>
#include<cstring>
using namespace std;

int main(){
    
    int n, i, j;
    scanf("%d", &n);

    for(i = 1; i <= n; i++){
        for(j = n - i; j > 0; j--)
            printf(" ");
        
        for(j = 1; j <= i * 2 - 1; j++)
            printf("*");
        puts("");
    }
    
    for(i = n - 1; i > 0; i--){
        for(j = n - i; j > 0; j--)
            printf(" ");
        
        for(j = 1; j <= i * 2 - 1; j++)
            printf("*");
        
        puts("");
    }
}


```
