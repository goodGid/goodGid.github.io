string solution(string s) {
    string answer = "";
    int size = s.size();
    if(size & 1){ // Odd
        answer = s[ size / 2];
    }else { // Even
        answer = s[ size / 2 - 1] ;
        answer += s[size/2];
    }
    return answer;
}

    string answer = "";

    int len = s.length();
    if (len%2 == 0) {
        answer = s.substr((len/2)-1, 2);

        answer = s[len/2];
    }
