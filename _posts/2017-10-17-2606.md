## Code
``` cpp


#include<iostream>
using namespace std;
int p[105];

int Find(int x ){
    if ( x == p[x])
        return x;
    else{
         int y = Find(p[x]);
         p[x] = y;
         return y;
        // return p[x] = Find(p[x]);
    }
    return x;
}

void Union(int x, int y){
    x = Find(x);
    y = Find(y);
    p[y] = x;
    
}

int main(){
    for(int i=1; i<=105; i++)
        p[i] = i;
    int n,m;
    cin >> n >> m;
    
    int a, b;
    while (m--) {
        scanf("%d%d", &a, &b);
        Union(a, b);
    }
    
    int ans=0;
    for(int i=1; i<=n; i++)
        if( Find(1) == Find(i))
            ans ++;
            
    cout << ans-1 << endl;
    return 0;
}
```
